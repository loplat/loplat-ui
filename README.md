## Installation
```sh
// with yarn
yarn add loplat-ui

// with npm
npm install loplat-ui
```

## Usage
```jsx
import React from 'react';
import { Button } from 'loplat-ui';

function Example() {
  return (
    <Button>Hello world!</Button>
  )
}
```
