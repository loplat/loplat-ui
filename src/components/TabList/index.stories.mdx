import { TabList } from './index';
import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs';
import { runTest } from './storybookTest.ts';
import { useState } from 'react';

<Meta title="Components/TabList" component={TabList} />

export const Template = ({ tabs, ...args }) => {
  const [selectedValue, setSelectedValue] = useState(tabs[0].value);
  return (
    <TabList
      {...args}
      tabs={tabs}
      selectedValue={selectedValue}
      onChange={(e, newValue) => setSelectedValue(newValue)}
      aria-label="loplat UI tablist"
    />
  );
};
