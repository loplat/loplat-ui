export const Template = (args) => {
  const [checked, setChecked] = useState(false);
  const toggle = (e) => {
    setChecked(e.target.checked);
  };
  return <Checkbox checked={checked} {...args} onChange={toggle} />;
};

export const Group = (args) => {
  const checkBoxList = [
    false,
    true,
    'checked',
    'unchecked',
    'intermediate',
    'unchecked',
    'checked',
    'intermediate',
  ].map((checked, idx) => ({
    checked,
    disabled: idx > checked.length / 2,
    children: `${checked}${idx > checked.length / 2 ? ' & disabled' : ''}`,
    name: idx,
  }));
  const [list, setList] = useState(checkBoxList);
  const toggle = (children, isChecked) => {
    const newResult = [...list].map((item) => {
      if (item.children === children) return { ...item, checked: isChecked };
      return item;
    });
    setList(newResult);
  };
  return (
    <>
      {list.map((props) => (
        <Checkbox {...props} onChange={(e) => toggle(props.children, e.target.checked)} key={props.name} mb={2} />
      ))}
    </>
  );
};
